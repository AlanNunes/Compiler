variable    : KEYWORD IDENTIFIER EQ expr
            : IDENTIFIER EQ expr
            : KEYWORD IDENTIFIER
expr	    : KEYWORD:declare IDENTIFIER EQ expr
comp-expr   : arith-expr (and|or) arith-expr
arith-expr  : expr (>|<|>=|<=|==) expr
            : term (PLUS|MINUS) term
term	    : factor (MUL|DIV) factor
factor	    : INT|FLOAT
power	    : factor ^ factor
